295  void update()
296  {
297  // Compute the normalised normal of [a, b] segment.
298  // Note: we do it here just to reduce a little bit the inter-dependencies.
299  auto const normalisedNormal = [](sf::Vector2f const& a, sf::Vector2f const& b) -> sf::Vector2f
300  {
301  sf::Vector2f const v = a - b;
302  sf::Vector2f const n = { v.y, -v.x };
303  float const length = std::sqrt(n.x * n.x + n.y * n.y);
304 
305  return length != 0.0 ? n / length : n;
306  };

308  m_vertices.clear();
309  m_vertices.resize(m_info.pointCount * 2 - 1); // Count points for the curve and (count - 1) points for the outline
310  // (that is, one point for the outline between each point of the curve)
311 
312  // Compute the points of the curve
313  float const step = (m_info.end - m_info.begin) / m_info.pointCount;
314  for (unsigned int i = 0; i < m_info.pointCount; ++i) {
315  float const t = m_info.begin + step * i;
316  m_vertices[i * 2] = { m_info.f(t), m_color }; // one vertex over two because of the outline
317  }
318 
319  // Compute the points of the outline
320  for (unsigned int i = 1; i < m_info.pointCount * 2 - 1; i += 2) {
321  // Compute the normal of the two points (p0 and p2) of the curve that are next to this outline point (p1)
322  sf::Vector2f const& p0 = m_vertices[i - 1].position;
323  sf::Vector2f const& p2 = m_vertices[i + 1].position;
324  sf::Vector2f const n = normalisedNormal(p0, p2);
325  sf::Vector2f const m = (p0 + p2) / 2.f;
326  sf::Vector2f const p1 = m + n * m_thickness;
327 
328  m_vertices[i] = { p1, m_color };
329  }
330  }
